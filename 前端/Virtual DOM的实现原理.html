# Virtual DOM的实现原理

# 为什么要使用Virtural DOM

  • 虚拟DOM可以维护程序的状态，跟踪上一次状态
  • 通过比较前后两次状态的差异更新真实的DOM

虚拟DOM的作用

  • 维护视图和状态的关系
  • 复杂视图情况下提升渲染
  • 除了渲染DOM以外，还可以实现SSR、原生应用和小程序等

Vitrual DOM库

  • Snabbdom (opens new window)
  • virtual-dom

# 模块

Snabbdom的核心库并不能处理元素的属性、样式、事件等，如果需要处理的话，可以引用模块

官方提供了6个模块

  • attributes

    • 设置DOM元素的属性，使用setAttribute
    • 处理布尔类型数据

  • props

    • 和attributes模块相似，设置DOM元素属性element[attr] = value
    • 不处理布尔类型数据

  • class

    • 切换类样式
    • 注意：给元素设置样式是通过sel选择器

  • dataset

    • 设置data-*的自定义事件

  • style

    • 设置行内样式，支持动画
    • delayed/remove/destory

  • enenvlisteners

    • 注册和移除事件

export function h(sel: any, b?: any, c?: any): VNode {
  var data: VNodeData = {}
  var children: any
  var text: any
  var i: number
  // 处理参数 实现函数的重载

  //处理三个参数 
  if (c !== undefined) {
    if (b !== null) {
      data = b
    }
    // 如果c是数组，说明c是子元素
    if (is.array(c)) {
      children = c
    }
    //如果c是原始值 （数字或字符串） 
    else if (is.primitive(c)) {
      text = c

    }
    // 判断c是VNode
    else if (c && c.sel) {
      children = [c]
    }
  }
  // 两个参数
  else if (b !== undefined && b !== null) {
    if (is.array(b)) {
      children = b
    } else if (is.primitive(b)) {
      text = b
    } else if (b && b.sel) {
      children = [b]
    } else { data = b }
  }
  if (children !== undefined) {
    // 处理children中的原始值
    for (i = 0; i < children.length; ++i) {
      // 如果children是string或number,创建vnode节点描述文本节点
      if (is.primitive(children[i])) children[i] = vnode(undefined, undefined, undefined, children[i], undefined)
    }
  }

  //如果是svg,添加命名空间 
  if (
    sel[0] === 's' && sel[1] === 'v' && sel[2] === 'g' &&
    (sel.length === 3 || sel[3] === '.' || sel[3] === '#')
  ) {
    addNS(data, children, sel)
  }
//   返回vnode
  return vnode(sel, data, children, text, undefined)
};

# vnode

export interface VNode {
  // 选择器
  sel: string | undefined
  // 节点数据： 属性、时间、样式等
  data: VNodeData | undefined
  // 子节点  和text互斥
  children: Array<VNode | string> | undefined
  // 记录VNode对应的真实DOM
  elm: Node | undefined
  // 节点中的内容 和children互斥  
  text: string | undefined
  // 优化用
  key: Key | undefined
}

export function vnode (sel: string | undefined,
  data: any | undefined,
  children: Array<VNode | string> | undefined,
  text: string | undefined,
  elm: Element | Text | undefined): VNode {
  const key = data === undefined ? undefined : data.key
  return { sel, data, children, text, elm, key }
}

# patch的整体过程

patch(oldVnode,newVnode)
  • 打补丁，把新节点中变化的内容渲染到真实DOM，最后返回新节点作为下一次处理的旧节点
  • 对比新旧VNode的是否是相同节点（sel和key是否相同）
  • 如果不是相同节点，删除之前的节点重新渲染
  • 是相同节点，就判断新的VNode是否有text，如果有并且和oldVNode的text不同，直接更新文本内容
  • 如果新的VNode有children,判断子节点是否有变化，判断子节点的过程使用的就是diff算法
  • diff过程只进行同层级比较

  let i: number
  let j: number
  const cbs: ModuleHooks = {
    create: [],
    update: [],
    remove: [],
    destroy: [],
    pre: [],
    post: []
  }

  // 初始化转换虚拟节点的api

  // 最终构建的cbs对象形式cbs = { create: [fn1,fn2], update: [fn3,fn4]}
  for (i = 0; i < hooks.length; ++i) {

  return function patch (oldVnode: VNode | Element, vnode: VNode): VNode {
      ...
  }
}

# patch

// 对比新旧VNode差异，把VNode渲染成真实DOM,并返回vnode
  return function patch (oldVnode: VNode | Element, vnode: VNode): VNode {
    let i: number, elm: Node, parent: Node
    // 保存新插入节点的队列，为了触发钩子函数
    const insertedVnodeQueue: VNodeQueue = []
    // 执行模块中的pre钩子函数
    for (i = 0; i < cbs.pre.length; ++i) cbs.pre[i]()

    // 判断oldVnode 是不是VNode，创建vnode并设置elm 
    if (!isVnode(oldVnode)) {
      // 把DOM元素转化成空的VNode  将sel属性设置为真实DOM
      oldVnode = emptyNodeAt(oldVnode)
    }

    // 判断新旧节点是相同节点（key和sel相同）
    if (sameVnode(oldVnode, vnode)) {
      // 找节点差异并更新DOM
      patchVnode(oldVnode, vnode, insertedVnodeQueue)
    } else {
      // 如果新旧节点不同 VNode创建对应的DOM

      // 获取oldVnode对应的真实DOM元素
      elm = oldVnode.elm!
      parent = api.parentNode(elm) as Node
      // 创建vnode对应的DOM元素，并触发用户传入的init/create钩子函数,vnode.elm赋给真实DOM
      createElm(vnode, insertedVnodeQueue)

       // 如果父节点不为空，把vnode对应的DOM插入到文档中
      if (parent !== null) {
      // 把vnode对应的DOM插入到文档中
        api.insertBefore(parent, vnode.elm!, api.nextSibling(elm))
        // 移除老节点
        removeVnodes(parent, [oldVnode], 0, 0)
      }
    }

    // 执行insert钩子函数
    for (i = 0; i < insertedVnodeQueue.length; ++i) {
      insertedVnodeQueue[i].data!.hook!.insert!(insertedVnodeQueue[i])
    }
    // 执行模块中的post钩子函数
    for (i = 0; i < cbs.post.length; ++i) cbs.post[i]()
    return vnode
  }
}

# createElm

// 创建vnode对应的DOM元素，并触发用户传入init/create钩子函数和模块中的create钩子函数
  // vnode.elm赋给真实DOM
 function createElm (vnode: VNode, insertedVnodeQueue: VNodeQueue): Node {
    let i: any
    let data = vnode.data
    // 执行用户传入的init钩子函数
    if (data !== undefined) {
      const init = data.hook?.init
      if (isDef(init)) {
        init(vnode)
        // 为什么要重新赋值，因为用户可能会在init钩子函数中重新修改data的值
        data = vnode.data
      }
    }
    const children = vnode.children
    const sel = vnode.sel
    // 如果是!创建注释节点
    if (sel === '!') {
      if (isUndef(vnode.text)) {
        vnode.text = ''
      }
      vnode.elm = api.createComment(vnode.text!)
    } else if (sel !== undefined) {
      // 解析选择器
      const hashIdx = sel.indexOf('#')
      const dotIdx = sel.indexOf('.', hashIdx)
      const hash = hashIdx > 0 ? hashIdx : sel.length
      const dot = dotIdx > 0 ? dotIdx : sel.length
      const tag = hashIdx !== -1 || dotIdx !== -1 ? sel.slice(0, Math.min(hash, dot)) : sel

      // 创建标签元素
      const elm = vnode.elm = isDef(data) && isDef(i = data.ns)
        ? api.createElementNS(i, tag)
        : api.createElement(tag)

      // 如果选择器有id或class,就会给创建的标签元素添加id或class
      if (hash < dot) elm.setAttribute('id', sel.slice(hash + 1, dot))
      if (dotIdx > 0) elm.setAttribute('class', sel.slice(dot + 1).replace(/\./g, ' '))
      // 执行模块中的create钩子函数
      for (i = 0; i < cbs.create.length; ++i) cbs.create[i](emptyNode, vnode)
      // 创建所有子节点对应的dom对象，并追加到elm上
      if (is.array(children)) {
        for (i = 0; i < children.length; ++i) {
          const ch = children[i]
          if (ch != null) {
            api.appendChild(elm, createElm(ch as VNode, insertedVnodeQueue))
          }
        }
      } else if (is.primitive(vnode.text)) {
        // 如果vnode是string或number，创建文本节点并追加到DOM树
        api.appendChild(elm, api.createTextNode(vnode.text))
      }
      // 执行用户传入的create钩子
      const hook = vnode.data!.hook
      if (isDef(hook)) {
        hook.create?.(emptyNode, vnode)
        if (hook.insert) {
          insertedVnodeQueue.push(vnode)
        }
      }
    } else {
      // 选择器为空，创建文本节点
      vnode.elm = api.createTextNode(vnode.text!)
    }
    return vnode.elm
  }

# addVnodes和removeVnodes

function createRmCb(childElm: Node, listeners: number) {
    return function rmCb() {
      // 没执行一次listeners就会-1
      if (--listeners === 0) {
        const parent = api.parentNode(childElm) as Node
        api.removeChild(parent, childElm)
      }
    }
  }

 function removeVnodes(parentElm: Node,
    vnodes: VNode[],
    startIdx: number,
    endIdx: number): void {
    for (; startIdx <= endIdx; ++startIdx) {
      let listeners: number
      let rm: () => void
      // vnodes的每一个元素
      const ch = vnodes[startIdx]
      if (ch != null) {
        // ch.sel有值
        if (isDef(ch.sel)) {
          // 执行用户和模块中的所有destroy函数（包含所有子节点的destroy钩子函数）
          invokeDestroyHook(ch)


          listeners = cbs.remove.length + 1
          // 创建删除的回调函数
          rm = createRmCb(ch.elm!, listeners)


          // 执行模块中的remove钩子函数
          for (let i = 0; i < cbs.remove.length; ++i) cbs.remove[i](ch, rm)
          // 获取用户传入的remove钩子函数
          const removeHook = ch?.data?.hook?.remove
          if (isDef(removeHook)) {
            removeHook(ch, rm)
          } else {
            // 如果用户没有传入remove钩子函数，直接调用删除元素的方法
            rm()
          }
        } else { // Text node
          //  ch.sel没有值  就是文本节点，直接删除文本节点
          api.removeChild(parentElm, ch.elm!)
        }
      }
    }
  }

addVnodes

function addVnodes(
    parentElm: Node,
    before: Node | null,
    vnodes: VNode[],
    startIdx: number,
    endIdx: number,
    insertedVnodeQueue: VNodeQueue
  ) {
    for (; startIdx <= endIdx; ++startIdx) {
      const ch = vnodes[startIdx]
      if (ch != null) {
        api.insertBefore(parentElm, createElm(ch, insertedVnodeQueue), before)
      }
    }
  }

# patchVnode

执行过程

  1. 触发prepatch钩子函数

  2. 触发update钩子函数

  3. 判断新节点是否有text属性

    1. 新节点有text属性，且不等于旧节点的text属性

      1. 如果老节点有children， 就移除children对应的DOM元素
      2. 设置新节点对应的DOM元素的textContext
    1. 新节点没有text属性

  • 新老节点都有children 调用updateChildren

  • 只有新节点有children

    • 如果老节点有text属性，清空对应DOM元素textContent
    • 添加所有的子节点 addVnodes
      • 只有老节点有children 移除老节点
      • 只有老节点有text属性 清空对应DOM的textContent
// oldVnode和vnode key和sel属性相同
  function patchVnode(oldVnode: VNode, vnode: VNode, insertedVnodeQueue: VNodeQueue) {
    const hook = vnode.data?.hook
    // 执行用户设置的prepatch钩子函数
    hook?.prepatch?.(oldVnode, vnode)
    const elm = vnode.elm = oldVnode.elm!
    const oldCh = oldVnode.children as VNode[]
    const ch = vnode.children as VNode[]
    // 新旧vnode相同直接返回
    if (oldVnode === vnode) return

    // 执行模块中的update钩子函数
    if (vnode.data !== undefined) {
      for (let i = 0; i < cbs.update.length; ++i) cbs.update[i](oldVnode, vnode)
      vnode.data.hook?.update?.(oldVnode, vnode)
    }
    // vnode中没有text属性
    if (isUndef(vnode.text)) {
      // oldVnode和vnode的子节点是否定义
      if (isDef(oldCh) && isDef(ch)) {
        // oldVnode和vnode的子节点 不相同
        // 使用diff算法对比子节点，更新子节点
        if (oldCh !== ch) updateChildren(elm, oldCh, ch, insertedVnodeQueue)
      }
      // vnode有子节点，oldVnode 没有子节点
       else if (isDef(ch)) {
      //oldVnode有text属性，  清空老节点的dom元素内容
      // 添加vnode中的所有子节点
        if (isDef(oldVnode.text)) api.setTextContent(elm, '')
        addVnodes(elm, null, ch, 0, ch.length - 1, insertedVnodeQueue)
      }
      // vnode没有子节点，oldVnode 有子节点
      else if (isDef(oldCh)) {
        // 移除oldVnode的所有子节点
        removeVnodes(elm, oldCh, 0, oldCh.length - 1)
      }
      // 如果oldVnode有text属性 直接清空
       else if (isDef(oldVnode.text)) {
        api.setTextContent(elm, '')
      }
    }
     // 如果设置了text属性  新旧两个vnode的text属性不相等，就更新
    else if (oldVnode.text !== vnode.text) {
    //  如果老节点存在子节点children,就移除
      if (isDef(oldCh)) {
        removeVnodes(elm, oldCh, 0, oldCh.length - 1)
      }
      api.setTextContent(elm, vnode.text!)
    }
    // 执行用户传入的postpatch钩子函数
    hook?.postpatch?.(oldVnode, vnode)
  }

# updateChildren

# updateChildren整体分析

  • 对比树的差异，我们可以取第一棵树的每一个节点与第二棵树的每一个节点进行比较，这样的时间复杂度是O(n^3)
  • 在DOM操作的时候我们很少会把一个父节点移动或更新到某一个子节点
  • 因此只需要找同级别的子节点进行比较，然后再找下一节点进行比较，这样的复杂度是O(n)

![img](.assets\Virtual DOM的实现原理\v2-e2ff9253f30633efcf123a4c6865abec_720w.jpg)

执行过程

  • oldStartVnode/newStartVnode

    • 新旧节点从开始位置开始比较
    • 旧节点第一个元素与新节点第一个元素进行使用sameVnode比较，如果相同调用pathVnode对比和更新节点
    • 比较完成后索引后移，往后进行比较

![img](.assets\Virtual DOM的实现原理\v2-6742b579f403ebbc7c946a6b1c9c2e61_720w.jpg)

  • oldEndVnode/newEndVnode（效果与上面相似，只不过是从结束位置开始比较，如果相同，比较完成后索引左移）

    • 新旧节点从结束位置开始比较
    • 如果相同调用pathVnode对比和更新节点
    • 比较完成后索引前移,往前移动进行比较

  • oldStartVnode/newEndVnode

    • 旧节点第一个节点、新节点最后一个节点开始比较
    • 如果相同就把如果相同调用pathVnode对比和更新节点，再把旧节点的第一个节点移动到最后
    • 旧节点索引后移，新节点索引前移，进行比较

![img](.assets\Virtual DOM的实现原理\v2-61c8aad6f4a9a0fd0990bd983b67b8dc_720w.jpg)

  • oldEndVnode/newStartVnode

    • 旧节点第一个节点、新节点的第一个节点开始比较
    • 如果相同就把如果相同调用pathVnode对比和更新节点，再把旧节点的最后一个节点移动到最前面
    • 更新索引，旧节点索引前移，新节点索引后移

![img](.assets\Virtual DOM的实现原理\v2-032fc415041f4bec5c503b47d6acae89_720w.jpg)

  • 如果上面四种情况都不满足

    • 在老节点中查找子节点newStartVnode对应的vnode

    • 没有就创建newStartVnode对应DOM并插入到oldStartVnode对应的DOM的前面

    • 存在，比较newStartVnode在老节点中对应的vnode elmToMove的sel属性是否相同

      • 不相同，直接创建（同上）
      • 相同，调用patchVnode，再将elmToMove对应的DOM插入到oldStartVnode对应的DOM前面

![img](.assets\Virtual DOM的实现原理\v2-40816dee4c2789017a9c521609a63c75_720w.jpg)

  • 上面执行完毕后，如果新老节点至少有一个没有遍历完成
  • 如果老节点先遍历完成，说明新节点没有遍历完，创建新节点还没有遍历的节点（从newStartIdx,newEndIdx）

![img](.assets\Virtual DOM的实现原理\v2-89490750a5cec603c8d2f9f21b5301f2_720w.jpg)

  • 否则说明老节点有多余的节点，删除老节点还没有遍历的节点（oldStartIdx, oldEndIdx）

function updateChildren(parentElm: Node,
    oldCh: VNode[],
    newCh: VNode[],
    insertedVnodeQueue: VNodeQueue) {
    let oldStartIdx = 0
    let newStartIdx = 0
    let oldEndIdx = oldCh.length - 1
    let oldStartVnode = oldCh[0]
    let oldEndVnode = oldCh[oldEndIdx]
    let newEndIdx = newCh.length - 1
    let newStartVnode = newCh[0]
    let newEndVnode = newCh[newEndIdx]
    let oldKeyToIdx: KeyToIndexMap | undefined
    let idxInOld: number
    let elmToMove: VNode
    let before: any

    while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
      // 判断新旧节点是否为null  (移动索引 新旧开始节点可能为null)
      if (oldStartVnode == null) {
        oldStartVnode = oldCh[++oldStartIdx] // Vnode might have been moved left
      } else if (oldEndVnode == null) {
        oldEndVnode = oldCh[--oldEndIdx]
      } else if (newStartVnode == null) {
        newStartVnode = newCh[++newStartIdx]
      } else if (newEndVnode == null) {
        newEndVnode = newCh[--newEndIdx]
      } 
      // 新旧开始节点进行比较
      else if (sameVnode(oldStartVnode, newStartVnode)) {
        patchVnode(oldStartVnode, newStartVnode, insertedVnodeQueue)
        oldStartVnode = oldCh[++oldStartIdx]
        newStartVnode = newCh[++newStartIdx]
      }
      // 新旧结束节点比较
      else if (sameVnode(oldEndVnode, newEndVnode)) {
        patchVnode(oldEndVnode, newEndVnode, insertedVnodeQueue)
        oldEndVnode = oldCh[--oldEndIdx]
        newEndVnode = newCh[--newEndIdx]
      }
      // oldStartVnode newEndVnode
      // 如果旧开始节点和新结束节点是同一节点
      else if (sameVnode(oldStartVnode, newEndVnode)) { // Vnode moved right
        patchVnode(oldStartVnode, newEndVnode, insertedVnodeQueue)
        // 旧结束节点对应的DOM元素  移动到最后
        api.insertBefore(parentElm, oldStartVnode.elm!, api.nextSibling(oldEndVnode.elm!))

        oldStartVnode = oldCh[++oldStartIdx]
        newEndVnode = newCh[--newEndIdx]
      } 
      // 如果旧结束节点和新开始节点是同一节点
      else if (sameVnode(oldEndVnode, newStartVnode)) { // Vnode moved left
        patchVnode(oldEndVnode, newStartVnode, insertedVnodeQueue)
          // 旧结束节点对应的DOM 移动到最前面
        api.insertBefore(parentElm, oldEndVnode.elm!, oldStartVnode.elm!)
        oldEndVnode = oldCh[--oldEndIdx]
        newStartVnode = newCh[++newStartIdx]
      }


      else {
        if (oldKeyToIdx === undefined) {
          //包含所有旧节点的对象 key为旧节点vnode的key，index为老节点的所在索引 {key:index}
          oldKeyToIdx = createKeyToOldIdx(oldCh, oldStartIdx, oldEndIdx)
        }

        // 在老节点上找与新节点newStartVnode key值相同的节点 的索引
        idxInOld = oldKeyToIdx[newStartVnode.key as string]
        if (isUndef(idxInOld)) { // New element
          // 如果索引不存在   旧将新节点的元素对应的DOM元素插入到老的开始节点的前面
          api.insertBefore(parentElm, createElm(newStartVnode, insertedVnodeQueue), oldStartVnode.elm!)
        } else {
          //如果key存在

          // 获取对应老节点的vNode
          elmToMove = oldCh[idxInOld]
          // key相同但是sel不同  说明该节点修改过
          if (elmToMove.sel !== newStartVnode.sel) {
            // 创建新的Vnode的真实DOM并插入到老节点的前面
            api.insertBefore(parentElm, createElm(newStartVnode, insertedVnodeQueue), oldStartVnode.elm!)
          } else {

            patchVnode(elmToMove, newStartVnode, insertedVnodeQueue)
            // 老节点赋值给undefined
            oldCh[idxInOld] = undefined as any
            // 把老节点移动到老的开始节点前面
            api.insertBefore(parentElm, elmToMove.elm!, oldStartVnode.elm!)
          }
        }
        newStartVnode = newCh[++newStartIdx]
      }
    }
    if (oldStartIdx <= oldEndIdx || newStartIdx <= newEndIdx) {
      // 如果老节点先遍历完成，说明新节点没有遍历完，创建新节点还没有遍历的节点
      if (oldStartIdx > oldEndIdx) {
        before = newCh[newEndIdx + 1] == null ? null : newCh[newEndIdx + 1].elm
        addVnodes(parentElm, before, newCh, newStartIdx, newEndIdx, insertedVnodeQueue)
      } else {
        // 删除老节点还没有遍历的节点
        removeVnodes(parentElm, oldCh, oldStartIdx, oldEndIdx)
      }
    }
  }
