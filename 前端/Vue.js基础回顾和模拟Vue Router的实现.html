# 动态路由

路由组件传参 (opens new window) 写法一

const User = {
  template: '<div>User</div>'
}
const router = new VueRouter({
  routes: [
    // 动态路径参数 以冒号开头
    { path: '/user/:id', component: User }
  ]
})

const User = {
  template: '<div>User {{ $route.params.id }}</div>'
}

写法二

const User = {
  template: '<div>User</div>'
}
const router = new VueRouter({
  routes: [
    // 动态路径参数 以冒号开头
    { 
        path: '/user/:id',
        component: User,
        // 如果 props 被设置为 true，route.params 将会被设置为组件属性。
        // props三种传参方式文档  https://router.vuejs.org/zh/guide/essentials/passing-props.html
        props:true 
    }
  ]
})

const User = {

# 嵌套路由

嵌套路由文档 (opens new window)

const router = new VueRouter({
  routes: [
    { path: '/user/:id', component: User,
      children: [
        {
          // 当 /user/:id 匹配成功，
          path: '',
          component: UserProfile
        },
        {
          // 当 /user/:id/profile 匹配成功，
          // UserProfile 会被渲染在 User 的 <router-view> 中
          path: 'profile',
          component: UserProfile
        },
        {
          // 当 /user/:id/posts 匹配成功
          // UserPosts 会被渲染在 User 的 <router-view> 中
          path: 'posts',
          component: UserPosts
        }
      ]
    }
  ]
})

# 编程式导航

# router.push(location, onComplete?, onAbort?)

该方法的参数可以是一个字符串路径，或者一个描述地址的对象

// 字符串
router.push('home')

// 对象
router.push({ path: 'home' })

// 命名的路由
router.push({ name: 'user', params: { userId: '123' }})

// 带查询参数，变成 /register?plan=private
router.push({ path: 'register', query: { plan: 'private' }})

# router.replace(location, onComplete?, onAbort?)

跟 router.push 很像，唯一的不同就是，它不会向 history 添加新记录，而是跟它的方法名一样 —— 替换掉当前的 history 记录。

# router.go(n)

这个方法的参数是一个整数，意思是在 history 记录中向前或者后退多少步，类似 window.history.go(n)。

# Hash和History模式

这两种模式都是客户端路由的实现方式，当路径发生变化，不会向服务器发送请求，由JS监视路径的变化，根据不同地址渲染不同内容

# 区别

# 表现形式的区别

hash模式 https://music.163.com/#playlist?id=...

History模式 https://music.163.com/playlist?id=... ,这种模式还需要服务端支持

# 原理的区别

  • Hash 模式是基于锚点，以及onhashchange事件

  • History 模式是基于HTML5中的History API

    • history.pushState() ie10以后才支持 与history的区别是history.push会向服务端发送请求，而history.pushState不会
    • history.replaceState()

# History模式的使用

在单页面（http://www.xxx.com）应用中，不存在（http://www.xxx.com/login）,直接访问服务器会报找不到该页面，所以服务端应该除了静态资源外部都返回单页应用的首页index.html

# Hash模式

  • url中的#后面的内容作为路径地址
  • 监听hashchage事件
  • 根据当前路由地址找到相应的组件重新渲染 #### Vue Router原理实现 History (opens new window)
  • 通过history.pushState方法改变地址栏（只改变地址）
  • 监听popstate事件（浏览器历史操作变化，pushState和relpaceState并不会触发该事件，浏览器的前进或后退、history.back或history.forward该事件才会被触发）
  • 监听当前路由地址找到相应的组件进行渲染

# Vue Router 模拟实现

let _Vue
class Router {
    static install(Vue) {
        if (Router.install.installed) {
            return
        }
        // 将Vue存在在当前
        _Vue = Vue
        Vue.mixin({
            beforeCreate() {
                if (this.$options.router) {
                    Vue.prototype.$router = this.$options.router
                    this.$options.router.init()
                }
            }
        })
    }
    constructor(options) {
        this.options = options
        this.mode = options.mode || 'hash'
        this.routerMap = {}
        this.data = _Vue.observable({
            current: '/'
        })
    }
    init() {
        this.initRouterMap()
        this.initComponent(_Vue)
        this.initCurrent()
        this.initEvent()
    }
    initRouterMap() {
        this.options.routes.map(route => {
            this.routerMap[route.path] = route.component
        })
    }
    initComponent(Vue) {
        const Link = {
            name: 'RouterLink',
            props: {
                to: {
                    type: String,
                    required: true
                }
            },
            methods: {}
        }
        const View = {
            render: h => h(this.routerMap[this.data.current])
        }
        if (this.mode === 'history') {
            Link.render = function (h) {
                return h('a', {
                    attrs: {
                        href: this.to
                    },
                    on: {
                        click: this.clickHandler
                    }
                }, this.$slots.default)
            }
            Link.methods.clickHandler = function (e) {
                e.preventDefault()
                history.pushState({}, '', this.to)
                this.$router.data.current = this.to
            }
        } else {
            Link.render = function (h) {
                return h('a', {
                    attrs: {
                        href: '/#' + this.to
                    }
                }, this.$slots.default)
            }
        }
        Vue.component('router-link', Link)
        Vue.component('router-view', View)
    }
    // 第一次加载页面初始化
    initCurrent() {
        // history模式
        if (this.mode === 'history') {
            this.data.current = location.pathname
            return
        }
        // hash模式
        if (location.hash === '') {
            window.location.hash = '#/'
        }
        this.data.current = location.hash.slice(1)
    }
    initEvent() {
        // 监听hash改变
        window.addEventListener('hashchange', () => {
            this.data.current = location.hash.slice(1)
        })
        // 监听浏览器的的历史记录  当历史记录条目更改时，将触发popstate事件
        window.addEventListener('popstate', () => {
            this.data.current = location.pathname
        })
    }
}
export default Router
