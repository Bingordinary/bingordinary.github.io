# 模块化规范

# 模块化规范的出现

CommonJS规范 每个文件就是一个模块 每个模块都有单独的作用域 通过module.exports导出成员 通过require函数载入模块 CommonJS是以同步模式加载模块

AMD Require.js

// 定义一个模块
define('module1', ['jquery','./module2'], function ($,module2) {
    return {
        start: function(){
            ...
        }
    }
})
// 加载一个模块
require(['./module1'], function () {
    module1.start()
})

目前，绝大多数第三方库都支持AMD规范

  • AMD使用起来比较复杂
  • 模块js文件请求频繁

Sea.js + CMD

define(function (require, exports, module) {
    // require引入依赖
    var $ = require('jquery')
    // exports或module.exports导出
    module.exports = function () {
        console.log('123')
    }
})

# ES Modules

# ES Modules特性

目前绝大多数浏览器都是支持ES Modules

ESM的基本特性

  • ESM自动采用严格模式，忽略'use strict'
  • 每个ES Module都是运行在单独的私有作用域中
  • ESM通过CORS的方式请求外部js模块的
  • ESM的script标签会延迟脚本执行
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <div>
        <div></div>
    </div>
    <!-- 通过给script添加 type = module 的属性，就可以以ES Module的标准执行其中的js代码了 -->
    <script type="module">
        // 1、ESM自动采用严格模式，忽略'use strict'
    console.log(this)//undefined
</script>

    <!-- 2、每个ES Module都是运行在单独的私有作用域中 -->
    <script type="module">
        var a = 100
</script>
    <script type="module">
        console.log(a)//报错
</script>
    <!--3、 ESM通过CORS的方式请求外部js模块的 -->
    <!-- 会报跨域 -->
    <script src="https://libs.qq.com/css/dialog-plus-min.js" type="module"></script>
    <script type="module" src="https://cdn.bootcdn.net/ajax/libs/vue/2.6.11/vue.min.js"></script>

    <!-- 4、 ESM的script标签会延迟脚本执行  相当于defer-->
    <script type="module">
        alert(0)
</script>
    <p>1231</p>
</body>

</html>

# ES Modules导出和导入

const foo = 'es modules'
export {
    foo
}
import {
    foo
} from './module.js'
console.log(foo)

注意事项 导出的成员并不字面量对象，导入的语法并不是对象的解构 导出的成员并不是成员的值，而是成员的引用关系 * 导入的成员是只读的，不能被修改

# 导入的用法

  • 导入模块必须填写完成路径

    • 不能省略文件的拓展名
    • 载入index.js也必须填写完整的路径
    • 相对路径./不能省略，可以使用'/'开头的绝对路径，也可以使用完整的url

  • 加载模块并不提取模块中的成员import {} from './module1.js

  • 使用import *导入需要载入模块的所有成员。import * as a from './module1.js

  • 动态加载模块.import(...).then()

  • 同时导出命名成员和默认成员。import { name ,default as a}='./module.js

// 加载模块
import {} from './module1.js
// 等价于
import from './module1.js


// 导入需要载入模块的所有成员
import * as a from './module1.js


// 动态加载模块
import('./module1.js').then(function (module){
    console.log(module)
})

// 同时导出命名成员和默认成员 
import { name ,default as a}='./module.js
// 等价于
import a,{ name }='./module.js

# 导入导出成员

import {
    Text
}
from './text.js'
export {Text}
export {
    Text,
    default as DetaultText
}
from './text.js'

# ES Modules在nodejs支持情况

  • 文件的拓展名为.mjs
  • 启动node加上--experimental-modules
node --experimental-modules app.mjs

在node14版本直接使用node app.mjs也不会有问题

// 会报错，因为导入不是对象的结构
import {
    camelCase
} from 'lodash'

// 系统内置模块没问题
import {
    writeFileSync
} from 'fs'
writeFileSync('./foo.txt', 'es module123')

# ES Modules与CommonJs交互

  • ES Modules中可以载入CommonJs模块
  • CommonJs 不能载入ES Modules 模块
  • CommonJs 始终都会导出一个默认成员
  • 注意import不是结构导出对象

app.mjs

import {
    name
} from './common.js'
console.log(name)

common.js

// 正常
import common from './common.js'
// 报错
import {
    name
} from './common.js'
console.log(name)

node环境不能在CommonJs模块中通过requre载入模块

const module1 = require('./module.mjs')
