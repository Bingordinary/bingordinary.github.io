# 手写Promise源码

# Promise类核心逻辑实现

  1. Promise就是一个类 在执行这个类的时候，需要传递一个执行器进去，执行器会立即执行
  2. Promise 中有三个状态分别为成功 fulfilled 失败rejected 等待pedding
  • pedding ->fulfilled
  • pedding ->rejected
  • 一旦状态确定就不可更改
  1. resolve和rejectd 使用来更改状态
  2. then 方法内部做的事情就是判断状态 如果状态是成功，调用成功的回调函数 如果状态是失败，调用失败回调
  3. then成功回调有一个参数 表示成功之后的值；失败回调有一个参数 表示失败的原因
const PENDING = 'pending'
const FULLFILLED = 'fulfilled'
const REJECTED = 'rejected'
class MyPromise {
    // promise状态
    status = PENDING
    value = undefined
    reson = undefined
    constructor(execuctor) {
        execuctor(this.resolve, this.reject)
    }
    resolve = (val) => {
        // 如果状态不是pending 阻止程序向下执行
        if (this.status !== PENDING) {
            return
        }
        this.status = FULLFILLED
        this.value = val
    }
    reject = (reson) => {
        if (this.status !== PENDING) {
            return
        }
        this.status = REJECTED
        this.reson = reson
    }
    then = (successCallback, failCallback) => {
        if (this.status === FULLFILLED) {
            successCallback(this.value)
        } else if (this.status === REJECTED) {
            failCallback(this.reson)
        }
    }
}

# Promomise实现加入异步逻辑

上面实现的promise ，在测试代码中加入异步代码无法正常运行

let promise = new MyPromise((resolve, reject) => {
    setTimeout(() => {
        resolve('data')
    }, 2000)
})

promise.then(val => {
    console.log('success')
    console.log(val)
}, err => {
    console.log('err')
    console.log(err)

原因是因为then方法在执行的的时候，传入的执行器立即执行，执行器的异步代码还没有执行完毕，then方法就已经开始执行了 。上面代码中还需要加入pending状态

# 实现then方法多次调用添加多个处理函数

当同一个promise多次调用then方法时 同步 会立即执行then方法的successCallback或successCallback 异步 将所以异步方法存起来，依次调用

...
    // 成功回调
    successCallback = []
    // 失败回调
    failCallback = []
 resolve = (val) => {
        // 如果状态不是pending 阻止程序向下执行
        if (this.status !== PENDING) {
            return
        }
        this.status = FULLFILLED
        this.value = val
        // 存在成功回调就执行
        // this.successCallback && this.successCallback(val)
        // 一次调用回调函数
        while (this.successCallback.length) {
            this.successCallback.shift()(val)
        }
    }
    reject = (reson) => {
        if (this.status !== PENDING) {
            return
        }
        this.status = REJECTED
        this.reson = reson
        // 存在失败回调就执行
        // this.failCallback && this.failCallback(reason)
        while (this.failCallback.length) {
            this.failCallback.shift()(reson)
        }
    }
    then = (successCallback, failCallback) => {
        if (this.status === FULLFILLED) {
            successCallback(this.value)
        } else if (this.status === REJECTED) {
            failCallback(this.reson)
        } else {
            // 等待  将成功回调和失败回调存储起来
            this.successCallback.push(successCallback)
            this.failCallback.push(failCallback)
        }
    }

# 实现then方法的链式调用

then方法是可以被链式调用的，后面的then方法的回调函数拿到的值是上一个then方法的回调函数

then(successCallback, failCallback) {
        const newPromise = new MyPromise((resolve, reject) => {
            if (this.status === FULLFILLED) {
                let res = successCallback(this.value)

                   resolve(res)
            } else if (this.status === REJECTED) {
                failCallback(this.err)
            } else {
                this.successCallback.push(successCallback)
                this.failCallback.push(failCallback)
            }
        })
        return newPromise
    }

此外，还需要根据then函数的返回值 判断返回值和newPromise对象是否是同一对象，是 判断返回值时promsie对象还是普通值 - 普通值直接resolve - promise对象 查看promise对象返回的结果 再根据结果决定调用resolve,还是reject

then(successCallback, failCallback) {
        const newPromise = new MyPromise((resolve, reject) => {
            if (this.status === FULLFILLED) {
                let res = successCallback(this.value)
                resolvePromise(res, resolve, reject)
            } else if (this.status === REJECTED) {
               failCallback(this.err)
            } else {
                this.successCallback.push(()=>{
                    let res = successCallback()
                    resolvePromise(res, resolve, reject)
                })
                this.failCallback.push(()=>{
                    failCallback()
                })
            }
        })
        return newPromise
}

function resolvePromise(res, resolve, reject) {
    if (res instanceof MyPromise) {
        res.then(resolve, reject)
    } else {
        resolve(res)
    }

# then方法链式调用识别返回的Promise对象

...
// then方法
const newPromise = new MyPromise((resolve, reject) => {
            // setTimeout为了确保new MyPromise执行完
            setTimeout(() => {
                // 成功 同步
                if (this.status === FULLFILLED) {
                    // 成功回调的返回值  代码执行出错，直接调用下一个函数reject传递给下一个then的回调函数
                    try {
                        let res = successCb(this.value)
                        resolvePromise(newPromise, res, resolve, reject)
                    } catch (e) {
                        reject(e)
                    }
                } else if (this.status === REJECTED) { //失败
                    try {
                        let res = failCb(this.error)
                        // 实现失败回调的返回值传入到下一个then回调函数中 
                        resolvePromise(newPromise, res, resolve, reject)
                    } catch (e) {
                        reject(e)
                    }
                } else {
                    // 等待 将成功回调和失败回调存起来  异步任务
                    this.successCb.push(() => {
                        try {
                            let res = successCb(this.value)
                            resolvePromise(newPromise, res, resolve, reject)
                        } catch (e) {
                            reject(e)
                        }
                    })
                    this.failCb.push(() => {
                        try {
                            let res = failCb(this.error)
                            resolvePromise(newPromise, res, resolve, reject)
                        } catch (e) {
                            reject(e)
                        }
                    })
                }
            }, 0)

        })
        return newPromise

function resolvePromise(newPromise, res, resolve, reject) {
    // 判断成功回调返回的值是不是跟当前MyPromise对象自己
    if (newPromise === res) {
        return reject(new Error('Uncaught (in promise) TypeError: Chaining cycle detected for promise #<Promise>'))
    }

    // 识别返回的promise对象
    if (res instanceof MyPromise) {
        res.then(resolve, reject)
    } else {
        // 普通值
        resolve(res)
    }
}

# 实现Promise.all

按照异步代码执行的顺序得到异步代码执行的结果 接收一个数组 一个失败直接reject,

...
  static all(arr) {
        let res = []
        let index = 0


        return new MyPromise((resolve, reject) => {
            function addData(key, data) {
                res[key] = data
                index++
                if (index === arr.length) {
                    resolve(res)
                }
            }
            for (let i = 0; i < arr.length; i++) {
                let current = arr[i]
                if (current instanceof MyPromise) {
                    current.then(res => {
                        addData(i, res)
                    }, err => {
                        // 一个出错直接reject
                        reject(err)
                    })
                } else {
                    addData(i, arr[i])
                }
            }
        })

# finally方法的实现

finally()方法用于指定不管 Promise 对象最后状态如何，都会执行的操作

finally(callback) {
        return this.then(val => {
            return MyPromise.resolve(callback()).then(() => val)
        }, err => {

            return MyPromise.resolve(callback()).then(() => {
                throw err
            })
        })
    }
